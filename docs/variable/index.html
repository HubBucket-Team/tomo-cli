    public BabelConfigModuleEditor: ModuleEditor source 

    import {BabelConfigModuleEditor} from 'tomo-cli/src/utils/index.js'

    Create and edit a Babel.js configuration file with a fluent API

    Example:

    Extend module.exports content and prepend text to the top of the file
    const cfg = new BabelConfigModuleEditor();
await cfg
    .create()
    .extend({
        presets: [`'@babel/preset-env'`]
    })
    .prepend(`const {existsSync} = require('fs-extra');`)
    .commit();

    public PackageJsonEditor: JsonEditor source 

    import {PackageJsonEditor} from 'tomo-cli/src/utils/index.js'

    Create and edit a package.json manifest file with a fluent API

    Example:

    Create a new package.json
    const pkg = new PackageJsonEditor();
await pkg.create().commit();
    Create a new package.json and read its contents (chaining OK)
    const pkg = new PackageJsonEditor();
const contents = pkg.create().read();
    Extend a package.json
    const script = {test: 'jest --coverage'};
await pkg.extend({script}).commit();
    Create and extend a package.json without writing to disk (chaining OK)
    const script = {
    lint: 'eslint index.js -c ./.eslintrc.js'
};
await pkg
    .create(false)
    .extend({script}, false)
    .commit();

    Test:

    • Verify package.json editor can create and edit a package.json manifest file
